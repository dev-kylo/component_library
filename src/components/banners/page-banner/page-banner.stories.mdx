  landing: true,
  pagetitle: 'I am a Page Title',
  content: '',
  heading: '',
  text: 'A one stop shop for everything you need to know.',
  colourscheme: '',

      pagetitle="${ifDefined(args.pagetitle || undefined)}"

      colourscheme="${ifDefined(args.colourscheme || undefined)}"
      pagetitlecolours="${ifDefined(args.pagetitlecolours || undefined)}"
      bgimage="${ifDefined(args.bgimage || undefined)}"
      image="${ifDefined(args.image || undefined)}"
      imagewidth="${ifDefined(args.imagewidth || undefined)}"
      imageheight="${ifDefined(args.imageheight || undefined)}"
      video="${ifDefined(args.video || undefined)}"
    >${args.content}</page-banner>`}

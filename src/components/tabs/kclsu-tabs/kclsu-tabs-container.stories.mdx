# `<kclsu-tabs-container>`

## Content

### Example

```html
<tabs-container>
  <tab-header name="A" active slot="tab-headers">a is the first letter</tab-header>,
  <tab-content name="A" slot="tab-content">
    <p>A, from the proto-caananite Aleph, is the first letter in the alphabet</p>
  </tab-content>
</tabs-container>
```

### Try It

<Canvas isColumn={true} withSource={true} isExpanded={true}>
  <Story name="With Content" component="tabs-container" args={{
    active: 'A',

      { name: 'C', summary: 'C', content: 'For some reason, the Romans had it out for "G"' },

  }}>
    {args => html`
      <tabs-container>
        ${args.tabs.map(tab => html`
          <tab-header name="${tab.name}" ?active="${args.active === tab.name}" slot="tab-headers">${tab.summary}</tab-header>,
          <tab-content name="${tab.name}" slot="tab-content">
            <p>${tab.content}</p>
          </tab-content>
        `)}
      </tabs-container>
    `}
  </Story>
  <ArgsTable story="With Content" exclude="tabs"/>
</Canvas>
